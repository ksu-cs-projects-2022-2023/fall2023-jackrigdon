#remove stop words, and tokenize
def preprocess_text(texts):
    stop_words = set(stopwords.words('english'))
    lmtzr = WordNetLemmatizer()
    # porter_stemmer = PorterStemmer()
    
    processed_texts = []
    for doc in texts:
        # Tokenization and case normalization
        tokens = word_tokenize(doc.lower())
        # Removing non-alphabetic tokens and stopwords
        tokens = [word for word in tokens if word.isalpha() and word not in stop_words]
        # Lemmatization (or stemming)
        tokens = [lmtzr.lemmatize(word) for word in tokens]
        # Optionally: stemming (uncomment if you want to use stemming instead of lemmatization)
        # tokens = [porter_stemmer.stem(word) for word in tokens]
        # Removing short words
        tokens = [word for word in tokens if len(word) >= 3]
        processed_texts.append(tokens)
        
    return processed_texts

Section Three: Understanding Probabilistic Topic Modeling¶

a.) Text classification is a supervised learning task where the goal is assigning predefined labels or categories, while topic modeling is an unsupervised learning task that aims to discover abstract topics present in a collection of text documents

b.) The goal of topic modeling (LDA) is to represent documents as mixtures of topics and topics as mixtures of words. It aims to reveal hidden thematic structure within a large coallition of documents, providing a way to summarize, and understand large datasets.

c.) A Document-Topic matrix represents the distribution of topics accros documents within a corpus. Each row relates to a document, and each column relates to a topics.The values within the matrix represent the probability of each topic in each document.

d.) Term-topic matrix is the distribution of words accross topics. Each row relates to a term or word, and each column relates to a topic. The values within the matrix represent the probability of each word belonging to each topic.

import time

def compute_coherence_values(dictionary, corpus, texts, limit, start=2, step=3):
    coherence_values = []
    model_list = []
    for num_topics in range(start, limit, step):
        start_time = time.time()
        print(f"Training LDA model with {num_topics} topics...")
        
        model = LdaModel(corpus=corpus, id2word=dictionary, num_topics=num_topics, passes=20, random_state=100)
        model_list.append(model)
        
        coherencemodel = CoherenceModel(model=model, texts=texts, dictionary=dictionary, coherence='c_v')
        coherence = coherencemodel.get_coherence()
        coherence_values.append(coherence)
        
        end_time = time.time()
        print(f"Model with {num_topics} topics complete. Coherence Score: {coherence:.4f}. Time taken: {end_time - start_time:.2f}s\n")
        
    return model_list, coherence_values

def get_most_representative_docs(document_topics):
    most_representative_docs = [-1] * optimal_model.num_topics
    max_contributions = [0] * optimal_model.num_topics

    for doc_id, doc_topics in enumerate(document_topics):
        for topic_id, contribution in doc_topics:
            if contribution > max_contributions[topic_id]:
                max_contributions[topic_id] = contribution
                most_representative_docs[topic_id] = doc_id

    return most_representative_docs
